    -- TARGET PLAYER SEARCH
    local function getClosestPlayerInFOV()
        local closestTarget = nil
        local shortestDistance = Config.FOV_Radius
        local mousePos = UserInputService:GetMouseLocation()
        
        for _, v in ipairs(Players:GetPlayers()) do
            if v ~= localPlayer then
                local isWhitelisted = Config.Whitelist[v.Name:lower()] or Config.Whitelist[v.DisplayName:lower()]
                if not isWhitelisted and v.Character and v.Character:FindFirstChild(Config.TargetPart) then
                    local targetPart = v.Character[Config.TargetPart]
                    local humanoid = v.Character:FindFirstChildOfClass("Humanoid")
                    
                    if humanoid and humanoid.Health > 0 then
                        local screenPos, onScreen = camera:WorldToViewportPoint(targetPart.Position)
                        if onScreen then
                            local isVisible = true
                            if Config.WallCheckEnabled then
                                local origin = camera.CFrame.Position
                                local dir = (targetPart.Position - origin)
                                local params = RaycastParams.new()
                                params.FilterDescendantsInstances = {localPlayer.Character, v.Character}
                                params.FilterType = Enum.RaycastFilterType.Exclude
                                params.IgnoreWater = true
                                
                                local rayResult = workspace:Raycast(origin, dir, params)
                                if rayResult then isVisible = false end
                            end

                            if isVisible then
                                local distanceToMouse = (Vector2.new(screenPos.X, screenPos.Y) - mousePos).Magnitude
                                if distanceToMouse < shortestDistance then
                                    shortestDistance = distanceToMouse
                                    closestTarget = targetPart
                                end
                            end
                        end
                    end
                end
            end
        end
        return closestTarget
    end

    local function teleportToLocation(locData)
        local char = localPlayer.Character
        if not char then return end

        if Config.TPMode == "Vehicle" then
            local hum = char:FindFirstChildOfClass("Humanoid")
            if hum and hum.SeatPart then
                local vehicle = hum.SeatPart:FindFirstAncestorOfClass("Model")
                if vehicle and vehicle.PrimaryPart then
                    vehicle:SetPrimaryPartCFrame(CFrame.new(locData.VehPos))
                else
                    hum.SeatPart.CFrame = CFrame.new(locData.VehPos)
                end
            else
                showNotification("Teleport Failed", "Anda tidak berada dalam kendaraan!", 2)
            end
        else
            local root = char:FindFirstChild("HumanoidRootPart")
            if root then 
                root.CFrame = CFrame.new(locData.CharPos) 
            end
        end
    end

    local function updateESP()
        local anyESP = Config.ESP_Highlight or Config.ESP_Name or Config.ESP_Health or Config.ESP_Inventory or Config.ESP_Box
        
        for _, player in ipairs(Players:GetPlayers()) do
            if player ~= localPlayer then
                local char = player.Character
                local hum = char and char:FindFirstChildOfClass("Humanoid")
                local root = char and char:FindFirstChild("HumanoidRootPart")
                local head = char and char:FindFirstChild("Head")
                
                local isWhitelisted = Config.Whitelist[player.Name:lower()] or Config.Whitelist[player.DisplayName:lower()]

                if anyESP and char and hum and root and head and hum.Health > 0 then
                    local cache = getESP(player)
                    
                    local rootPos, onScreen = camera:WorldToViewportPoint(root.Position)
                    local headPos = camera:WorldToViewportPoint(head.Position + Vector3.new(0, 0.5, 0))
                    local legPos = camera:WorldToViewportPoint(root.Position - Vector3.new(0, 3, 0))
                    
                    if onScreen then
                        local height = math.abs(headPos.Y - legPos.Y)
                        local width = height / 2

                        -- 1. Hitbox (Blue Outline)
                        if Config.ESP_Box and not isWhitelisted then
                            cache.Box.Size = Vector2.new(width, height)
                            cache.Box.Position = Vector2.new(rootPos.X - width / 2, rootPos.Y - height / 2)
                            cache.Box.Visible = true
                        else
                            cache.Box.Visible = false
                        end

                        -- 2. Name & Item
                        if Config.ESP_Name then
                            local nameStr = isWhitelisted and ("[WHITELIST] " .. player.Name) or player.Name
                            if Config.ESP_Item then
                                local tool = char:FindFirstChildOfClass("Tool")
                                nameStr = nameStr .. "\n[" .. (tool and tool.Name or "Fists") .. "]"
                            end
                            cache.Name.Text = nameStr
                            cache.Name.Position = Vector2.new(rootPos.X, headPos.Y - 25)
                            cache.Name.Color = isWhitelisted and Color3.fromRGB(50, 255, 50) or Color3.fromRGB(255, 255, 255)
                            cache.Name.Visible = true
                        else
                            cache.Name.Visible = false
                        end

                        -- 3. Health Bar + Numb
                        if Config.ESP_Health then
                            local healthPercent = math.clamp(hum.Health / hum.MaxHealth, 0, 1)
                            local barHeight = height
                            local barY = rootPos.Y - height / 2
                            local barX = rootPos.X - width / 2 - 6

                            cache.HealthBg.From = Vector2.new(barX, barY)
                            cache.HealthBg.To = Vector2.new(barX, barY + barHeight)
                            cache.HealthBg.Visible = true

                            local healthColor = Color3.fromRGB(255 * (1 - healthPercent), 255 * healthPercent, 0)
                            cache.HealthVal.From = Vector2.new(barX, barY + barHeight)
                            cache.HealthVal.To = Vector2.new(barX, barY + barHeight - (barHeight * healthPercent))
                            cache.HealthVal.Color = healthColor
                            cache.HealthVal.Visible = true

                            cache.HealthNum.Text = tostring(math.floor(hum.Health))
                            cache.HealthNum.Position = Vector2.new(barX - 16, barY + barHeight - (barHeight * healthPercent) - 6)
                            cache.HealthNum.Color = healthColor
                            cache.HealthNum.Visible = true
                        else
                            cache.HealthBg.Visible = false
                            cache.HealthVal.Visible = false
                            cache.HealthNum.Visible = false
                        end

                        -- 4. Inventory List
                        if Config.ESP_Inventory then
                            local invStr = ""
                            local backpack = player:FindFirstChild("Backpack")
                            if backpack then
                                for _, item in ipairs(backpack:GetChildren()) do
                                    if item:IsA("Tool") then invStr = invStr .. item.Name .. "\n" end
                                end
                            end
                            
                            if invStr ~= "" then
                                cache.InvText.Text = "Inv:\n" .. invStr
                                cache.InvText.Position = Vector2.new(rootPos.X + width / 2 + 5, rootPos.Y - height / 2)
                                cache.InvText.Visible = true
                            else
                                cache.InvText.Visible = false
                            end
                        else
                            cache.InvText.Visible = false
                        end

                        -- 5. Highlight (Ungu)
                        if Config.ESP_Highlight and not isWhitelisted then
                            cache.Highlight.Adornee = char
                            cache.Highlight.Parent = CoreGui
                        else
                            cache.Highlight.Parent = nil
                        end

                    else
                        hideESP(player)
                    end
                else
                    hideESP(player)
                end
            end
        end
    end

    -- Loops Runtime Event
    RunService.RenderStepped:Connect(function(deltaTime)
        if not screenGui or not screenGui.Parent then return end

        local fps = math.floor(1 / deltaTime)
        globalFPSLabel.Text = "FPS: " .. tostring(fps)
        if fps > 120 then globalFPSLabel.TextColor3 = Color3.fromRGB(0, 255, 0)
        elseif fps >= 60 then globalFPSLabel.TextColor3 = Color3.fromRGB(255, 255, 0)
        else globalFPSLabel.TextColor3 = Color3.fromRGB(255, 0, 0) end

        fovCircle.Position = UserInputService:GetMouseLocation()
        fovCircle.Visible = Config.ShowFOVCircle

        updateESP()

        -- Aim Assist Logic
        local currentTarget = "None"
        if Config.AimAssistEnabled then
            local target = getClosestPlayerInFOV()
            if target then
                currentTarget = target.Parent.Name
                local targetCFrame = CFrame.lookAt(camera.CFrame.Position, target.Position)
                camera.CFrame = camera.CFrame:Lerp(targetCFrame, Config.AimSmoothness)
            end
        end

        -- Auto Lock Camera Saat Spectating
        if isSpectating and specTarget and specTarget.Character then
            local targetHum = specTarget.Character:FindFirstChild("Humanoid")
            if targetHum and targetHum.Health > 0 then
                workspace.CurrentCamera.CameraSubject = targetHum
            end
        end

        if tabs["SYSTEM DSR"].Visible then
            stat_FPS.Text = "FPS: " .. tostring(fps)
            stat_Target.Text = "Target: " .. currentTarget
            stat_Time.Text = "Server Time: " .. os.date("%X")
            stat_Players.Text = "Players: " .. tostring(#Players:GetPlayers()) .. "/" .. tostring(Players.MaxPlayers)
            pcall(function() stat_Ping.Text = "Ping: " .. tostring(math.floor(Stats.Network.ServerStatsItem["Data Ping"]:GetValue())) .. " ms" end)
            stat_Mem.Text = "Memory: " .. tostring(math.floor(Stats:GetTotalMemoryUsageMb())) .. " MB"
        end
    end)
end
